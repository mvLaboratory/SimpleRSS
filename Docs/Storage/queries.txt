CREATE TABLE `RssItem` (
	`Id`	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE,
	`PublicationdDate`	TEXT,
	`Title`	TEXT NOT NULL,
	`Author`	TEXT,
	`Url`	TEXT NOT NULL,
	`Category_Id`	INTEGER,
	`CreationDate`	TEXT NOT NULL
);

CREATE TABLE `Category` (
	`Id`	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE,
	`Name`	TEXT NOT NULL,
	`RssURL`	TEXT NOT NULL,
	`CreationDate`	TEXT NOT NULL
);
